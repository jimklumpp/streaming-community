Using

This component is available via maven. Include in your pom.xml file the following :-

    <project>
        ...
        <dependencies>
            <dependency>
                <groupId>com.tibco.sb.compexch</groupId>
                <artifactId>WebSocketClientSample</artifactId>
                <type>ep-liveview-fragment</type>
            </dependency>
            ...
        </dependencies>
        ...
        <dependencyManagement>
            <dependencies>
                <dependency>
                     <groupId>com.tibco.sb.compexch</groupId>
                     <artifactId>WebSocketClientSample</artifactId>
                     <version>1.0.0</version>
                 </dependency>
                 ...
             </dependencies>
        </dependencyManagement>
        ...
    </project>
