30  /**
31   * Connects to wikimedia's IRC servers for real-time wikipedia updates. Implements
32   * basic conenction management (no reconnect). Has hard coded assumptions about
33   * the message protocol and format, and is not a suitable general-purpose IRC adapter.
34   * <p>
35   * See {@link WikiEdit} for details on handling of messages
36   * </p>

80  	public void typecheck() throws TypecheckException {
81  		if (server.isEmpty()) {
82  			throw new PropertyTypecheckException("server", "IRC Server may not be empty");
83  		}
84  		if (channel.isEmpty()) {
85  			throw new PropertyTypecheckException("channel", "Channel may not be empty");
86  		}
87  		if (nick.isEmpty()) {
88  			throw new PropertyTypecheckException("nick", "IRC Nickname may not be empty");
89  		}
90  		Schema output = new Schema(null,
91  				FIELD_PAGENAME,
92  				FIELD_DIFFURL,
93  				FIELD_EDITOR,
94  				FIELD_LINEDELTA,
95  				FIELD_COMMENT);
96  		setOutputSchema(0, output);

100 	 * Initialize the adapter. If typecheck succeeds, the init method is called before

103 	 * initialization of a resource such as, for example, a JDBC pool.

105 	public void init() throws StreamBaseException {
106 		super.init();

108 		irc = new IRCApiImpl(false);
109 		// update output schema and field references now
110 		outputSchema = getOutputSchema(0);
111 		FIELD_PAGENAME = outputSchema.getField(FIELD_PAGENAME.getName());
112 		FIELD_DIFFURL = outputSchema.getField(FIELD_DIFFURL.getName());
113 		FIELD_EDITOR = outputSchema.getField(FIELD_EDITOR.getName());
114 		FIELD_LINEDELTA = outputSchema.getField(FIELD_LINEDELTA.getName());
115 		FIELD_COMMENT = outputSchema.getField(FIELD_COMMENT.getName());

124 	private void sendAsync(WikiEdit edit) throws StreamBaseException {
125 		if (edit != null) {
126 			Tuple t = outputSchema.createTuple();
127 			t.setField(FIELD_PAGENAME, edit.pageName);
128 			t.setField(FIELD_DIFFURL, edit.diffURL);
129 			t.setField(FIELD_EDITOR, edit.editor);
130 			t.setField(FIELD_LINEDELTA, edit.lineDelta);
131 			t.setField(FIELD_COMMENT, edit.comment);
132 			sendOutputAsync(0, t);
133 		}
134 	}

136 	/**
137 	 * This method will be called when an adapter starts or resumes execution,
138 	 * before any registered runnables are started or resumed.  Note that if the
139 	 * application is shutdown directly from a suspended state, this will
140 	 * <b>not</b> be called; instead {@link #shutdown()} will be called.
141 	 *
142 	 * resume() is a callback that is called by the StreamBase runtime.
143 	 */
144 	public void resume() {
145 		if (ircState != null && ircState.isConnected()) {
146 			logger.debug("Ignoring resume(); IRC state already established and connected");
147 			return;
148 		}
149 		
150         final CountDownLatch latch = new CountDownLatch(1); 
151         final Exception[] failureException = new Exception[1];
152 		irc.connect(new IServerParameters() {
153 			
154 			@Override
155 			public IRCServer getServer() {
156 				return new IRCServer(server, port);
157 			}
158 			
159 			@Override
160 			public String getRealname() {
161 				return realName;
162 			}
163 			
164 			@Override
165 			public String getNickname() {
166 				return nick;
167 			}
168 			
169 			@Override
170 			public String getIdent() {
171 				return System.getProperty("user.name"); //$NON-NLS-1$
172 			}
173 			
174 			@Override
175 			public List<String> getAlternativeNicknames() {
176 				List<String> result = new ArrayList<String>();
177 				for (int i = 2; i < 10; ++i) {
178 					result.add(getNickname() + i);
179 				}
180 				return result;
181 			}
182 		}, new Callback<IIRCState>() {
183 
184 			@Override
185 			public void onSuccess(IIRCState aObject) {
186 				ircState = aObject;
187 				latch.countDown();
188 			}
189 
190 			@Override
191 			public void onFailure(Exception aExc) {
192 				failureException[0] = aExc;
193 			}
194 			
195 		});
196 		// wait until connected
197         try {
198         	logger.debug("Connecting to IRC server {}...", server);
199             latch.await(getConnectTimeoutSecs(), TimeUnit.SECONDS);
200         } catch (InterruptedException e) {
201         	logger.debug("Interrupted awaiting connection to IRC server", e);
202         }
203         if (failureException[0] != null) {
204         	logger.error("Could not connect to IRC server", failureException[0]);
205         	sendErrorOutput(failureException[0]);
206         }
207         if (ircState != null && ircState.isConnected()) {
208         	logger.info("Connected to {}, registering listener and joining channel {}", ircState.getServer().getHostname(), channel);
209         	irc.addListener(this);
210         	irc.joinChannel(channel); // TODO async to guard against blocks here
211         	logger.debug("Channel joined");
212         } else {
213         	logger.warn("No connection established; resume adapter to try again");
214         }
215         
216 	}

220 	@Override
221 	public void onMessage(IMessage aMessage) {
222 		logger.trace("received {}", aMessage);
223 		if (isSuspended()) {
224 			logger.trace("suspended; message dropped");
225 			return;
226 		}
227 		if (aMessage instanceof ChannelPrivMsg) {
228 			ChannelPrivMsg msg = (ChannelPrivMsg) aMessage;
229 			try {
230 				sendAsync(WikiEdit.fromChannelMessage(msg));
231 			} catch (CannotParseAsWikiEditException e) {
232 				// given that we do not handle many kinds of messages, low level debug
233 				logger.trace("Could not parse channel message as Wiki edit", e);
234 			} catch (StreamBaseException e) {
235 				logger.warn("Error sending Wiki Edit tuple downstream", e);
236 			}
237 		}
238 	}
